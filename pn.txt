CREATE DATABASE ADVISOR_PROCTOR_MANGEMEENT;

CREATE TABLE COURSE(C_code INT,Credits INT,C_name VARCHAR(50),Hours INT,PRIMARY KEY(C_code),UNIQUE(C_name),
FOREIGN KEY(Credits)REFERENCES COURSE(C_code));





CREATE TABLE STUDENT(Fname VARCHAR(50) NOT NULL,Mint VARCHAR(10),Lname VARCHAR(10) NOT NULL,SSN INT NOT NULL,
Dob DATE NOT NULL,
Address VARCHAR(100) NOT NULL,
Section VARCHAR(10),Email_id VARCHAR(100) NOT NULL,C_code INT NOT NULL,PRIMARY KEY(SSN),FOREIGN KEY(C_code)REFERENCES COURSE(C_code));




CREATE TABLE PROCTOR(Fname VARCHAR(50),Minit VARCHAR(10),Lname VARCHAR(10),P_id INT NOT NULL,C_code INT,
I_office VARCHAR(50),
Email_id VARCHAR(50) NOT NULL,Section VARCHAR(15),PRIMARY KEY(P_id),UNIQUE(C_code),
FOREIGN KEY(P_id)REFERENCES STUDENT(SSN));




CREATE TABLE ADVISOR(Fname VARCHAR(50),Minit VARCHAR(10),Lname VARCHAR(20),Advisor_ID  CHAR(15) NOT NULL,
Section VARCHAR(10) NOT NULL,P_id VARCHAR(40),C_code INT,Email_id VARCHAR(100),
PRIMARY KEY (Advisor_ID) ,FOREIGN KEY(P_id)REFERENCES PROCTOR(P_id),
FOREIGN KEY(C_code)REFERENCES COURSE(C_code));




CREATE TABLE ISSUES(ST_name VARCHAR(20), ST_SSN INT,ST_Section VARCHAR(20),Course_ID INT,Sem VARCHAR(20),
Complaints CHARACTER(100),PRIMARY KEY(ST_SSN),FOREIGN KEY(ST_SSN)REFERENCES STUDENT(SSN),
FOREIGN KEY(C_code)REFERENCES COURSE(C_code));
